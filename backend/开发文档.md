# 多客陪玩平台 - 后端开发文档

## 一、项目概述

### 1.1 项目简介

多客陪玩平台后端是一个基于 Node.js + Express 的 RESTful API 服务，提供用户管理、聊天消息、礼物系统、订单管理、支付处理等核心功能。

### 1.2 技术栈

| 类别 | 技术 | 版本 |
|------|------|------|
| 运行时 | Node.js | 18.x+ |
| 框架 | Express | 4.x |
| ORM | Sequelize | 6.x |
| 数据库 | MySQL | 8.0 |
| 缓存 | Redis | 7.x |
| 文档 | Swagger | 3.x |
| 认证 | JWT | - |

#### 认证中间件 (auth.js)

```javascript
// src/middlewares/auth.js
const authMiddleware = async (req, res, next) => {
  try {
    const token = req.headers.authorization?.replace('Bearer ', '')

    if (!token) {
      return res.fail(401, '请先登录')
    }

    const decoded = jwt.verify(token, jwtConfig.secret)
    req.userId = decoded.userId
    req.userInfo = decoded

    next()
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.fail(401, '登录已过期')
    }
    return res.fail(401, '无效的Token')
  }
}

// 可选认证中间件（不强制登录）
const optionalAuth = async (req, res, next) => {
  try {
    const token = req.headers.authorization?.replace('Bearer ', '')
    if (token) {
      const decoded = jwt.verify(token, jwtConfig.secret)
      req.userId = decoded.userId
    }
  } catch (error) {
    // 忽略错误，继续处理
  }
  next()
}
```

```javascript
const userController = {
  // 获取用户信息
  async getUserInfo(req, res) {
    try {
      const userId = req.query.userId || req.userId

      const user = await User.findByPk(userId, {
        attributes: { exclude: ['password'] }
      })

      if (!user) {
        return res.fail(404, '用户不存在')
      }

      // 统计关注数和粉丝数
      const [followCount, fansCount] = await Promise.all([
        UserFollow.count({ where: { user_id: userId } }),
        UserFollow.count({ where: { follow_user_id: userId } })
      ])

      // 返回前端期望的字段格式
      const userInfo = {
        userId: user.id,
        nickName: user.nickname,
        avatar: user.avatar,
        level: user.lv,
        vip: user.vip,
        vipLevel: user.vip_lv,
        vipExpireTime: user.vip_expire_time,
        gender: user.sex === 1 ? 'male' : user.sex === 2 ? 'female' : 'unknown',
        region: user.city,
        balance: parseFloat(user.money),
        score: user.score,
        followCount,
        fansCount,
        status: user.status,
        createTime: user.create_time,
        lastLoginTime: user.last_login_time
      }

      return res.ok({ user: userInfo })
    } catch (error) {
      console.error('获取用户信息失败:', error)
      return res.fail(500, '服务器错误')
    }
  },

  // 更新用户信息
  async updateUserInfo(req, res) {
    try {
      const { nickName, avatar, sex, city, signature } = req.body

      await User.update({
        nickname: nickName,
        avatar,
        sex,
        city,
        dec: signature
      }, {
        where: { id: req.userId }
      })

      return res.ok(null, '更新成功')
    } catch (error) {
      console.error('更新用户信息失败:', error)
      return res.fail(500, '服务器错误')
    }
  },

  // 关注用户
  async follow(req, res) {
    try {
      const { userId } = req.body

      if (userId === req.userId) {
        return res.fail(400, '不能关注自己')
      }

      const targetUser = await User.findByPk(userId)
      if (!targetUser) {
        return res.fail(404, '用户不存在')
      }

      await UserFollow.create({
        user_id: req.userId,
        follow_user_id: userId
      })

      return res.ok(null, '关注成功')
    } catch (error) {
      console.error('关注用户失败:', error)
      return res.fail(500, '服务器错误')
    }
  },

  // 取消关注
  async unfollow(req, res) {
    try {
      const { userId } = req.body

      await UserFollow.destroy({
        where: {
          user_id: req.userId,
          follow_user_id: userId
        }
      })

      return res.ok(null, '取消关注成功')
    } catch (error) {
      console.error('取消关注失败:', error)
      return res.fail(500, '服务器错误')
    }
  },

  async getFans(req, res) {

      const offset = (page - 1) * pageSize

      const result = await UserFollow.findAndCountAll({

          model: User,
```

```javascript
  async getChatList(req, res) {
    try {
      const { page = 1, pageSize = 20 } = req.query

      const chatRooms = await ChatRoom.findAndCountAll({

          ]

          model: User,
          as: 'user1',

          model: User,
          as: 'user2',

        }],
        limit: parseInt(pageSize),
        offset,
        order: [['update_time', 'DESC']]
      })

      const list = chatRooms.rows.map(room => {
        const otherUser = room.user_id1 === req.userId ? room.user2 : room.user1
        return {
          roomId: room.id,
          targetUserId: otherUser.id,
          nickName: otherUser.nickname,
          avatar: otherUser.avatar,
          lastMessage: room.last_message,
          lastMessageTime: room.last_message_time,
          unreadCount: room.unread_count || 0
        }
      })

      return res.ok({
        list,
        total: chatRooms.count,
        page: parseInt(page),
        pageSize: parseInt(pageSize)
      })
    } catch (error) {
      console.error('获取聊天列表失败:', error)
      return res.fail(500, '服务器错误')
```

```javascript
// src/services/chatService.js
class ChatService {
  async createChatRoom(userId1, userId2) {
    const existingRoom = await ChatRoom.findOne({
      where: {
        [Op.or]: [
          { user_id1: userId1, user_id2: userId2 },
          { user_id1: userId2, user_id2: userId1 }
        ]
      }
    })

    if (existingRoom) {
      return existingRoom
    }

    return await ChatRoom.create({
      user_id1: Math.min(userId1, userId2),
      user_id2: Math.max(userId1, userId2),
      create_time: Date.now(),
      update_time: Date.now(),
      unread_count: 0
    })
  }

  async getMessageHistory(userId1, userId2, page = 1, pageSize = 20) {
    const messages = await ChatMessage.find({
      $or: [
        { fromUserId: userId1, toUserId: userId2 },
        { fromUserId: userId2, toUserId: userId1 }
      ]
    })
    .sort({ createTime: -1 })
    .skip((page - 1) * pageSize)
    .limit(pageSize)

    return messages.reverse()
  }
}

module.exports = new ChatService()
```

#### 通话计费服务 (callBillingService.js)

```javascript
// src/services/callBillingService.js
class CallBillingService {
  /**
   * 创建通话记录
   * @param {Object} params - 通话参数
   * @param {number} params.callerId - 主叫用户ID
   * @param {number} params.calleeId - 被叫用户ID
   * @param {number} params.type - 通话类型 1-语音 2-视频
   * @param {number} params.pricePerMin - 每分钟价格（金币）
   */
  async createCallRecord({ callerId, calleeId, type, pricePerMin }) {
    return await CallRecord.create({
      caller_id: callerId,
      callee_id: calleeId,
      type,
      price_per_min: pricePerMin,
      start_time: Date.now(),
      status: 1
    })
  }

  /**
   * 结束通话并结算费用
   * @param {number} recordId - 通话记录ID
   * @param {number} duration - 通话时长（秒）
   */
  async endCall(recordId, duration) {
    const record = await CallRecord.findByPk(recordId)
    const minutes = Math.ceil(duration / 60)
    const totalCost = minutes * record.price_per_min

    // 扣除主叫方余额
    await User.decrement('money', {
      by: totalCost,
      where: { id: record.caller_id }
    })

    // 增加被叫方收入
    await User.increment('money', {
      by: totalCost * 0.8, // 平台抽成20%
      where: { id: record.callee_id }
    })

    // 更新通话记录状态
    await record.update({
      duration,
      total_cost: totalCost,
      end_time: Date.now(),
      status: 2
    })

    // 记录计费明细
    await CallBilling.create({
      record_id: recordId,
      duration,

      total_amount: totalCost,
```

## 九、注意事项

1. **数据库连接池**: MySQL连接池默认最大10个连接，根据服务器配置调整
2. **索引优化**: 频繁查询的字段已添加索引，避免全表扫描
3. **事务处理**: 涉及多表操作时使用事务确保数据一致性
4. **日志记录**: 使用 `console.log` 记录日志到 `logs/` 目录
5. **安全建议**:
   - 生产环境务必修改JWT密钥
   - 敏感配置使用环境变量
   - 启用HTTPS
   - 定期备份数据库
6. **字段映射**: 用户信息返回时需注意字段映射（如 `nickname` -> `nickName`, `lv` -> `level`）
7. **货币单位**: 所有业务订单统一使用金币作为交易货币
